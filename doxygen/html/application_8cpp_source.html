387void Application::addHistory(const QString &filepath) {
391 bool alreadyExists = std::any_of(historyActions.begin(), historyActions.end(),
392 [&filepath](const QAction *action) {
393 return filepath == action->data().toString();
394 });
395
399 if (alreadyExists) {
400 return;
401 }
402
406 auto *action = new QAction(QFileInfo(filepath).fileName(), this);
407 action->setData(filepath);
408 connect(action, &QAction::triggered, [this, filepath]() {
409 controller->openFile(filepath); // 点击记录时打开对应文件
410 filename = filepath;
411 });
412
416 historyActions.prepend(action);
417 ui->menuHistory->insertAction(ui->menuHistory->actions().isEmpty() ? nullptr : ui->menuHistory->actions().first(),
418 action);
419
423 while (historyActions.size() > maxHistorySize) {
424 QAction *lastAction = historyActions.takeLast();
425 ui->menuHistory->removeAction(lastAction);
426 delete lastAction;
427 }
428
429 saveHistory();
430}
